## PIN 인증 시스템

### 작동 방식
```
페이지 진입 → PinScreen 표시 → 4자리 입력 → 일치 시 sessionStorage 저장 → 메인 진입
틀리면 → shake 애니메이션 + 빨간 점 + 자동 초기화
```

### 구현 코드
```typescript
// App.tsx
const CORRECT_PIN = '0000'  // 초기세팅 — 보안 상 변경 필요
const AUTH_KEY    = 'crypto_auth_v2'

function PinScreen({ onAuth }: { onAuth: () => void }) {
  const [pin, setPin]   = useState('')
  const [shake, setShake] = useState(false)
  const [error, setError] = useState(false)

  const handleKey = (k: string) => {
    if (k === 'del') { setPin(p => p.slice(0,-1)); setError(false); return }
    if (pin.length >= 4) return
    const next = pin + k
    setPin(next)
    if (next.length === 4) {
      setTimeout(() => {
        if (next === CORRECT_PIN) {
          sessionStorage.setItem(AUTH_KEY, '1')
          onAuth()
        } else {
          setShake(true); setError(true)
          setTimeout(() => { setShake(false); setPin(''); setError(false) }, 700)
        }
      }, 120)
    }
  }
  // 키패드: ['1','2','3','4','5','6','7','8','9','','0','del']
}

function App() {
  const [authed, setAuthed] = useState(
    () => sessionStorage.getItem(AUTH_KEY) === '1'
  )
  if (!authed) return <PinScreen onAuth={() => setAuthed(true)} />
  return <MainLayout /> // 탭 시스템
}
```

RSI 계산 함수:
function calcRSI(closes, period = 14) {
  const changes = closes.slice(1).map((c, i) => c - closes[i])
  const gains   = changes.map(c => c > 0 ? c : 0)
  const losses  = changes.map(c => c < 0 ? Math.abs(c) : 0)
  let avgGain = gains.slice(0, period).reduce((a,b)=>a+b) / period
  let avgLoss = losses.slice(0, period).reduce((a,b)=>a+b) / period
  const rsi = []
  for (let i = period; i < gains.length; i++) {
    avgGain = (avgGain * (period-1) + gains[i]) / period
    avgLoss = (avgLoss * (period-1) + losses[i]) / period
    rsi.push(100 - 100/(1 + avgGain/avgLoss))
  }
  return rsi
}

### 포지션 흐름 분류 알고리즘
```javascript
function classifyFlow(prevOI, currOI, frAvg) {
  const oiUp = currOI > prevOI
  const frPos = frAvg > 0.005  // 과열 기준 0.005%
  if (oiUp && frPos)   return 'long_entry'
  if (oiUp && !frPos)  return 'short_entry'
  if (!oiUp && frPos)  return 'long_exit'
  if (!oiUp && !frPos) return 'short_exit'
  return 'neutral'
}
```

function calcBearishScore(record) {
  let score = 0

  if (record.oiChange3d > 0.10)  score += 20  // 급등

### 예측 모델 (유사 패턴 매칭)
```javascript
// 유사도 = 코사인 유사도
function cosineSimilarity(a, b) {
  const dot = a.reduce((s, v, i) => s + v * b[i], 0)
  const magA = Math.sqrt(a.reduce((s, v) => s + v*v, 0))
  const magB = Math.sqrt(b.reduce((s, v) => s + v*v, 0))
  return dot / (magA * magB)
}

function findPatterns(current, history) {
  const currentVec = [current.oiChange3d, current.cumFr3d, current.priceChange3d]
  return history
    .map(h => ({
      date: h.date,
      similarity: cosineSimilarity(currentVec,
        [h.oiChange3d, h.cumFr3d, h.priceChange3d]),
      return3d:  (h.priceAfter3d - h.price) / h.price * 100,
      return7d:  (h.priceAfter7d - h.price) / h.price * 100,
      return14d: (h.priceAfter14d - h.price) / h.price * 100,
    }))
    .filter(h => h.similarity > 0.9)
    .sort((a,b) => b.similarity - a.similarity)
    .slice(0, 5)
}
```
